310 //Compares two cards if they are up
311 void Game_c::compare_cards()
312 {
313  //Getting all cards that are up
314  this->cards = this->deck.map_nodes_if<Deck::Node*>([](auto n){ return n; }, [](auto n){ return n->get_info()->is_up(); });
315 
316  //Flipping cards if are more than 2
317  if(this->cards.size() > 2)
318  for(auto n : cards)
319  n->get_info()->hide_card();
320 
321  //Comparing cards if 2 of them are up
322  if(this->cards.size() == 2)
323  {
324  Deck::Node* card1 = cards[0];
325  Deck::Node* card2 = cards[1];
326 
327  this->triesText->setPlainText(QString("Intentos: " + QString::number(++this->tries)));
328 
329  //If cards are equal, change opacity, selected cards, remove cards from deck and add 1 to pairs_found
330  if(this->deck.map_adjacent_arcs<Deck::Arc*>(card1, [](auto a){ return a; }).get_first()->get_connected_node(card1)->get_info()->get_name(1) == card2->get_info()->get_name(1))
331  {
332  //Changing opacity of selected identical cards
333  card1->get_info()->setOpacity(0.6);
334  card2->get_info()->setOpacity(0.6);
335 
336  //Disabling selected identical cards
337  card1->get_info()->setEnabled(false);
338  card2->get_info()->setEnabled(false);
339 
340  //Removing cards from the deck
341  this->deck.remove_node(card1);
342  this->deck.remove_node(card2);
343 
344  this->pairs_found++;
345  }
346  //If cards are diferent, wait 0.5 seconds and then hide the cards
347  else
348  {
349  QThread::msleep(500);
350  for(auto n : cards)
351  n->get_info()->hide_card();
352  }
353  }
354 
355  //If there are no more cards, stop timer and then display restart game scene
356  if(this->deck.get_num_nodes() == 0)
357  {
358  this->timer->stop();
359  this->displayRestartGame();
360  }
361 }
